\section{Appendix: Data Sources}
\label{sec:appendix_data_sources}

Currently two version for JSON files for data sources definitions are supported, the current and the deprecated format.

\subsection{Current Format}

\begin{lstlisting}
{
    "content_type": "data_sources",
    "content_version": "0.2",
    "data_sources": [
        {
            "ds_type": "Tracker",
            "name": "MRTS",
            "sac": 1,
            "sic": 3
        },
        {
            "ds_type": "Tracker",
            "name": "ARTAS",
            "sac": 1,
            "sic": 4
        },
        {
            "ds_type": "ADSB",
            "info": {
                "network_lines": {
                        "L1": {
                            "listen_ip": "10.16.9.1",
                            "mcast_ip": "239.19.7.24",
                            "mcast_port": 8600
                        },
                        "L3": {
                            "listen_ip": "10.15.9.1",
                            "mcast_ip": "239.19.7.25",
                            "mcast_port": 8600
                        }
                }
            },
            "name": "ADSB",
            "sac": 1,
            "short_name": "WADS",
            "sic": 5
        },
        {
            "ds_type": "MLAT",
            "info": {
                        "L3": {
                            "listen_ip": "10.16.9.1",
                            "mcast_ip": "239.19.7.24",
                            "mcast_port": 8600
                        },
                        "L4": {
                            "listen_ip": "10.15.9.1",
                            "mcast_ip": "239.19.7.25",
                            "mcast_port": 8600
                        }
                }
            },
            "name": "MLAT1",
            "sac": 50,
            "short_name": "MLAT1",
            "sic": 70
        }
    ]
}
\end{lstlisting}
